<Properties>

  <Property name="body" type="textarea">
    Body (JSON).
  </Property>
</Properties>

<Properties>
  <Property name="statusCode" type="number">HTTP status code.</Property>
  <Property name="body" type="object">Response body (JSON).</Property>
  <Property name="headers" type="object">Response headers.</Property>
</Properties>
